  function tounicode(data) {
        if (data == '') return '请输入汉字';
        var str = '';
        for (var i = 0; i < data.length; i++) {
            str += "\\u" + parseInt(data[i].charCodeAt(0), 10).toString(16);
        }
        return str;
    }
    console.log(tounicode('哎呦我的哥'));
